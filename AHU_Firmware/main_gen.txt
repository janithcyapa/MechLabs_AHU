#include <stdio.h>
#include "freertos/FreeRTOS.h"
#include "freertos/task.h"
#include "driver/gpio.h"
#include "driver/ledc.h" // Added to access the raw frequency changer
#include "util_pwm.hpp"

// ==========================================
// 1. PIN DEFINITIONS & CONFIG
// ==========================================
const gpio_num_t RELAY_PINS[6] = {
    GPIO_NUM_13, GPIO_NUM_16, GPIO_NUM_17, 
    GPIO_NUM_23, GPIO_NUM_32, GPIO_NUM_33
};

// RELAY LOGIC (Set to 1 for NO wiring that needs HIGH to turn on)
const uint32_t RELAY_ON_STATE = 1; 
const uint32_t RELAY_OFF_STATE = !RELAY_ON_STATE;

// ==========================================
// 2. PWM INSTANCES
// ==========================================
PwmUtil servo_1(18, LEDC_CHANNEL_0, LEDC_TIMER_0);
PwmUtil servo_2(19, LEDC_CHANNEL_1, LEDC_TIMER_0);

// FIXED: Swapped GPIO 25 and 27
PwmUtil led_red(27, LEDC_CHANNEL_2, LEDC_TIMER_1);   // Was 25, now 27
PwmUtil led_green(26, LEDC_CHANNEL_3, LEDC_TIMER_1); 
PwmUtil led_blue(25, LEDC_CHANNEL_4, LEDC_TIMER_1);  // Was 27, now 25

PwmUtil buzzer(4, LEDC_CHANNEL_5, LEDC_TIMER_2);

// ==========================================
// 3. HELPER FUNCTIONS
// ==========================================

// --- RELAYS ---
void init_relays() {
    for (int i = 0; i < 6; i++) {
        gpio_reset_pin(RELAY_PINS[i]);
        gpio_set_direction(RELAY_PINS[i], GPIO_MODE_OUTPUT);
        gpio_set_level(RELAY_PINS[i], RELAY_OFF_STATE); 
    }
}

void set_relay(int relay_num, bool state) {
    if (relay_num >= 1 && relay_num <= 6) {
        gpio_set_level(RELAY_PINS[relay_num - 1], state ? RELAY_ON_STATE : RELAY_OFF_STATE);
    }
}

// --- RGB LED (COMMON ANODE) ---
void set_rgb(float r_percent, float g_percent, float b_percent) {
    uint32_t r_duty = 8191 - (uint32_t)(r_percent * 8191.0 / 100.0);
    uint32_t g_duty = 8191 - (uint32_t)(g_percent * 8191.0 / 100.0);
    uint32_t b_duty = 8191 - (uint32_t)(b_percent * 8191.0 / 100.0);

    led_red.set_raw_duty(r_duty);
    led_green.set_raw_duty(g_duty);
    led_blue.set_raw_duty(b_duty);
}

// --- BUZZER (FIXED DYNAMIC PITCH) ---
void play_tone(uint32_t freq_hz, int duration_ms) {
    if (freq_hz > 0) {
        // Safely change the timer frequency without touching the GPIO config!
        // We know the buzzer uses LEDC_TIMER_2 from our definition above.
        ledc_set_freq(LEDC_LOW_SPEED_MODE, LEDC_TIMER_2, freq_hz);
        
        // Turn volume on (50% duty cycle)
        buzzer.set_raw_duty(4095); 
    } else {
        buzzer.set_raw_duty(0); 
    }
    
    vTaskDelay(pdMS_TO_TICKS(duration_ms));
    buzzer.set_raw_duty(0); 
}

// Custom Sounds
void play_sound_startup() {
    play_tone(1000, 100); vTaskDelay(pdMS_TO_TICKS(50));
    play_tone(1500, 100); vTaskDelay(pdMS_TO_TICKS(50));
    play_tone(2000, 200); 
}

void play_sound_error() {
    play_tone(250, 300); vTaskDelay(pdMS_TO_TICKS(100));
    play_tone(250, 300); 
}

void play_sound_alert() {
    for (int i = 0; i < 4; i++) {
        // Lowered frequencies to the buzzer's "loud" sweet spot
        // Increased beep duration to 100ms for more impact
        play_tone(1500, 100); 
        vTaskDelay(pdMS_TO_TICKS(50));
        
        play_tone(2000, 100); 
        vTaskDelay(pdMS_TO_TICKS(50));
    } 
}


// ==========================================
// 4. MAIN APPLICATION
// ==========================================
extern "C" void app_main(void) {
    // Initialize standard GPIOs
    init_relays();

    // Initialize PWM devices (Done exactly ONCE here)
    servo_1.init(50);
    servo_2.init(50);
    led_red.init(5000);
    led_green.init(5000);
    led_blue.init(5000);
    buzzer.init(2000);
    
    // Play startup chime
    play_sound_startup();
    vTaskDelay(pdMS_TO_TICKS(2000)); // Wait 2 seconds before entering loop

    while (1) {
        // --- 1. RGB LED Demo ---
        printf("LED: RED\n");
        set_rgb(100, 0, 0); 
        vTaskDelay(pdMS_TO_TICKS(3000)); // LARGE DELAY

        printf("LED: YELLOW\n");
        set_rgb(100, 100, 0); 
        vTaskDelay(pdMS_TO_TICKS(3000)); // LARGE DELAY
        
        printf("LED: OFF\n");
        set_rgb(0, 0, 0); 
        vTaskDelay(pdMS_TO_TICKS(2000)); // LARGE DELAY

        // --- 2. Buzzer & Relay Alert Demo ---
        printf("Triggering Alert Sound & Relay 1 ON...\n");
        set_relay(1, true); 
        play_sound_alert();
        
        vTaskDelay(pdMS_TO_TICKS(3000)); // LARGE DELAY - Watch the relay!
        
        printf("Turning Relay 1 OFF...\n");
        set_relay(1, false);
        vTaskDelay(pdMS_TO_TICKS(2000)); // LARGE DELAY

        // --- 3. Servo Demo ---
        printf("Moving Servos to 90 degrees...\n");
        servo_1.set_angle_180(90);
        vTaskDelay(pdMS_TO_TICKS(3000)); // LARGE DELAY

        printf("Moving Servos to 0 degrees...\n");
        servo_1.set_angle_180(0);
        vTaskDelay(pdMS_TO_TICKS(2000)); // LARGE DELAY
        
        // --- 4. Error Sound Demo ---
        printf("Playing Error Sound...\n");
        play_sound_error();

        printf("Loop finished. Restarting in 5 seconds...\n");
        printf("----------------------------------------\n");
        vTaskDelay(pdMS_TO_TICKS(5000)); // VERY LARGE DELAY before restarting
    }
}
