static const char *TAG = "AHU_MAIN";
static i2c_master_dev_handle_t mux_dev_handle = NULL;

// 1. Declare sensor objects globally so they persist in memory
static sens_aht21::AHT21 aht21_sensor;
static sens_bme280::BME280 bme280_sensor;
static sens_ens160::ENS160 ens160_sensor;

// Mux Initialization
esp_err_t init_mux(i2c_master_bus_handle_t bus) {
    i2c_device_config_t mux_cfg = {};
    mux_cfg.dev_addr_length = I2C_ADDR_BIT_LEN_7;
    mux_cfg.device_address = 0x70; 
    mux_cfg.scl_speed_hz = 100000; 
    
    return i2c_master_bus_add_device(bus, &mux_cfg, &mux_dev_handle);
}

// Mux Channel Selector
esp_err_t select_mux_channel(uint8_t channel) {
    if (channel > 7 || mux_dev_handle == NULL) return ESP_ERR_INVALID_ARG;
    uint8_t control_register = (1 << channel);
    return i2c_master_transmit(mux_dev_handle, &control_register, 1, -1);
}

// Unified Data Structure
struct CombinedSensorData {
    float temperature = 0.0f;
    float humidity = 0.0f;
    float pressure = 0.0f;
    uint16_t eco2 = 0;
    uint16_t tvoc = 0;
    uint8_t aqi = 0;
};

enum SensorType { TYPE_AHT21, TYPE_ENS160, TYPE_BME280 };

// Unified Reading Function (Reads ONLY, no initialization)
esp_err_t get_sensor_data(uint8_t channel, SensorType type, CombinedSensorData &out_data) {
    // Switch the Mux to the desired channel
    esp_err_t err = select_mux_channel(channel);
    if (err != ESP_OK) return err;

    // Perform the read based on type
    switch (type) {
        case TYPE_AHT21: {
            return aht21_sensor.read(out_data.temperature, out_data.humidity);
        }
        case TYPE_BME280: {
            sens_bme280::Bme280Data bme_raw;
            err = bme280_sensor.read_data(bme_raw);
            if (err == ESP_OK) {
                out_data.temperature = bme_raw.temperature;
                out_data.humidity = bme_raw.humidity;
                out_data.pressure = bme_raw.pressure;
            }
            return err;
        }
        case TYPE_ENS160: {
            sens_ens160::Ens160Data ens_raw;
            err = ens160_sensor.read_data(ens_raw);
            if (err == ESP_OK) {
                out_data.aqi = ens_raw.aqi;
                out_data.tvoc = ens_raw.tvoc;
                out_data.eco2 = ens_raw.eco2;
            }
            return err;
        }
    }
    return ESP_FAIL;
}

extern "C" void app_main(void)
{
    ESP_LOGI(TAG, "Starting AHU Controller...");

    // 1. Initialize the Master I2C bus
    i2c_util::i2c_init();
    i2c_master_bus_handle_t bus = i2c_util::get_bus_handle();
    i2c_util::i2c_scan();

    // 2. Initialize the Multiplexer
    ESP_ERROR_CHECK(init_mux(bus));

    // 3. Initialize Sensors ONCE
    ESP_LOGI(TAG, "--- Initializing Sensors ---");
    
    // Init BME280 on Channel 6
    select_mux_channel(6);
    if (bme280_sensor.init(bus) == ESP_OK) {
        ESP_LOGI(TAG, "BME280 Initialized successfully on CH6");
    } else {
        ESP_LOGE(TAG, "Failed to initialize BME280!");
    }

    // Init AHT21 and ENS160 on Channel 7
    select_mux_channel(7);
    if (aht21_sensor.init(bus) == ESP_OK) {
        ESP_LOGI(TAG, "AHT21 Initialized successfully on CH7");
    } else {
        ESP_LOGE(TAG, "Failed to initialize AHT21!");
    }
    
    if (ens160_sensor.init(bus) == ESP_OK) {
        ESP_LOGI(TAG, "ENS160 Initialized successfully on CH7");
    } else {
        ESP_LOGE(TAG, "Failed to initialize ENS160!");
    }
    
    ESP_LOGI(TAG, "--- Sensor Init Complete. Starting Main Loop ---");

    CombinedSensorData data;

    // 4. Main reading loop
    while (true) {
        // Read BME280 on Channel 6
        if (get_sensor_data(6, TYPE_BME280, data) == ESP_OK) {
            ESP_LOGI("APP", "CH6 BME280 | T: %.2f C | P: %.2f hPa", data.temperature, data.pressure);
        }

        // Read AHT21 on Channel 7
        if (get_sensor_data(7, TYPE_AHT21, data) == ESP_OK) {
            ESP_LOGI("APP", "CH7 AHT21  | T: %.2f C | H: %.2f %%", data.temperature, data.humidity);
            
            // Feed the latest Temp/Hum to the ENS160 for compensation!
            // Since we are already on Channel 7, we can just call it directly.
            ens160_sensor.set_environment(data.temperature, data.humidity);
        }

        // Read ENS160 on Channel 7
        if (get_sensor_data(7, TYPE_ENS160, data) == ESP_OK) {
            ESP_LOGI("APP", "CH7 ENS160 | eCO2: %d ppm", data.eco2);
        }

        vTaskDelay(pdMS_TO_TICKS(5000));
    }
}
